package "Domain Models" {

  abstract class User {
    +int Id
    +string Email
    +string FirstName
    +string LastName
    +string PasswordHash
    +UserRole Role
    +bool IsActive
    +DateTime CreatedAt
    +string GetFullName()
  }

  class Student {
    +bool InitialPasswordSent
    +bool HasCompletedSelection()
  }

  class Admin {
    +bool CanManageSystem
  }

  class Presentation {
    +int Id
    +string Topic
    +DateTime StartsAt
    +int RoomId
    +int MaxObservers = 6
    +bool HasFreeSeats()
    +bool IsAtSameTime(Presentation other)
  }

  class Room {
    +int Id
    +string Name
  }

  class Registration {
    +int Id
    +DateTime RegisteredAt
    +int StudentId
    +int PresentationId
  }

  enum UserRole {
    Student
    Admin
  }

  User <|-- Student
  User <|-- Admin

  Room "1" -- "0..*" Presentation
  Student "1" -- "0..2" Registration
  Presentation "1" -- "0..6" Registration
}

package "Data Access" {

  class ApplicationDbContext {
    +DbSet<User> Users
    +DbSet<Student> Students
    +DbSet<Admin> Admins
    +DbSet<Presentation> Presentations
    +DbSet<Room> Rooms
    +DbSet<Registration> Registrations
    +SaveChanges()
  }

  ApplicationDbContext --> User
  ApplicationDbContext --> Student
  ApplicationDbContext --> Admin
  ApplicationDbContext --> Presentation
  ApplicationDbContext --> Room
  ApplicationDbContext --> Registration
}

package "Services" {

  class AuthService {
    +User Login(string email, string password)
    +string HashPassword(string password)
    +bool VerifyPassword(string password, string hash)
    +void RegisterFailedLogin(string email)
  }

  class StudentService {
    +List<Student> GetAllStudents()
    +Student GetStudentById(int id)
    +void CreateStudent(Student student)
    +void UpdateStudent(Student student)
    +void DeleteStudent(int id)
    +string GenerateInitialPassword()
  }

  class PresentationService {
    +List<Presentation> GetAllPresentations()
    +List<Presentation> GetAvailablePresentations(Student student)
    +Presentation GetPresentationById(int id)
    +void CreatePresentation(Presentation presentation)
    +void UpdatePresentation(Presentation presentation)
    +void DeletePresentation(int id)
  }

  class RegistrationService {
    +List<Registration> GetRegistrationsByStudent(int studentId)
    +void CreateRegistration(int studentId, int presentationId)
    +void DeleteRegistration(int registrationId)
    +bool CanRegister(Student student, Presentation presentation)
  }

  class AssignmentRuleService {
    +bool HasMaximumTwoRegistrations(Student student)
    +bool HasNoTimeConflict(Student student, Presentation presentation)
    +bool HasFreeSeats(Presentation presentation)
    +bool IsSelectionComplete(Student student)
  }

  class CsvImportService {
    +void ImportStudents(Stream csvFile)
    +void ImportPresentations(Stream csvFile)
  }

  class NotificationService {
    +void SendCredentials(Student student, string password)
    +void SendConfirmation(Student student)
    +void SendAdminNotification(Admin admin)
  }

  class ExportService {
    +byte[] ExportRoomObserverListAsCsv()
    +byte[] ExportRoomObserverListAsPdf()
    +RoomObserverList CreatePrintList()
  }

  class SystemResetService {
    +void ResetSchoolYear()
  }

  AuthService --> ApplicationDbContext
  StudentService --> ApplicationDbContext
  PresentationService --> ApplicationDbContext
  RegistrationService --> ApplicationDbContext
  RegistrationService --> AssignmentRuleService
  CsvImportService --> StudentService
  CsvImportService --> PresentationService
  NotificationService --> Student
  NotificationService --> Admin
  ExportService --> ApplicationDbContext
  SystemResetService --> ApplicationDbContext
}
